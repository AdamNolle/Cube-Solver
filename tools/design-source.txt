  applyOrbit(rx, ry) {

    this.cubeGroup.quaternion.premultiply(qy).premultiply(qx);

  buildCube(N) {
    if (this.cubies) this.cubies.forEach(c => this.cubeGroup.remove(c.mesh));
    this.cubies = [];
    if (this.core) { this.cubeGroup.remove(this.core); this.core = null; }
    if (this.faceBox) { this.cubeGroup.remove(this.faceBox); this.faceBox = null; }

    if (this.mode === 'texture') this.buildTextureCube(N);
    else this.buildCubieCube(N);

  }

  buildCubieCube(N) {
    const T = window.THREE;
    const total = 2.5;
    const step = total / N;
    const cubie = step * 0.95;
    this.step = step; this.half = (N - 1) / 2;
    // solid dark core so the gaps never reveal the far side
    const core = new T.Mesh(new T.BoxGeometry(total * 0.9, total * 0.9, total * 0.9), this.bodyMat);
    core.castShadow = false; core.receiveShadow = false;
    this.cubeGroup.add(core); this.core = core;
    const bodyGeo = new T.BoxGeometry(cubie, cubie, cubie);
    const stGeo = new T.PlaneGeometry(cubie * 0.84, cubie * 0.84);
    const eps = cubie * 0.5 + step * 0.012;
    const M = { px: 5, nx: 4, py: 0, ny: 1, pz: 2, nz: 3 };
    for (let ix = 0; ix < N; ix++) for (let iy = 0; iy < N; iy++) for (let iz = 0; iz < N; iz++) {
      const shell = ix === 0 || ix === N - 1 || iy === 0 || iy === N - 1 || iz === 0 || iz === N - 1;
      if (!shell) continue;
      const mesh = new T.Mesh(bodyGeo, this.bodyMat);
      const shadow = N <= 10;
      mesh.castShadow = shadow; mesh.receiveShadow = shadow;
      mesh.position.set((ix - this.half) * step, (iy - this.half) * step, (iz - this.half) * step);
      const add = (matIdx, nx, ny, nz, rotAxis, rotAng) => {
        const st = new T.Mesh(stGeo, this.faceMats[matIdx]);
        st.position.set(nx * eps, ny * eps, nz * eps);
        if (rotAxis === 'x') st.rotation.x = rotAng;
        if (rotAxis === 'y') st.rotation.y = rotAng;
        st.castShadow = false;
        mesh.add(st);
      };
      if (ix === N - 1) add(M.px, 1, 0, 0, 'y', Math.PI / 2);
      if (ix === 0) add(M.nx, -1, 0, 0, 'y', -Math.PI / 2);
      if (iy === N - 1) add(M.py, 0, 1, 0, 'x', -Math.PI / 2);
      if (iy === 0) add(M.ny, 0, -1, 0, 'x', Math.PI / 2);
      if (iz === N - 1) add(M.pz, 0, 0, 1, 'y', 0);
      if (iz === 0) add(M.nz, 0, 0, -1, 'y', Math.PI);
      mesh.userData.idx = [ix, iy, iz];
      this.cubeGroup.add(mesh);
      this.cubies.push({ mesh });
    }
  }

  buildTextureCube(N) {
    const T = window.THREE;
    const total = 2.5;
    const d = Math.min(N, 96);
    const solvedIdx = [5, 4, 0, 1, 2, 3]; // px,nx,py,ny,pz,nz -> r,o,w,y,g,b
    this.texFaces = [];
    const mats = [];
    for (let f = 0; f < 6; f++) {
      const cv = document.createElement('canvas'); cv.width = cv.height = 512;
      const tex = new T.CanvasTexture(cv);
      tex.anisotropy = 4;
      const face = { cv, ctx: cv.getContext('2d'), tex, d, solvedColor: this.faceList[solvedIdx[f]], scr: [] };
      for (let i = 0; i < d * d; i++) face.scr.push(this.faceList[Math.floor(Math.random() * 6)]);
      this.texFaces.push(face);
      mats.push(new T.MeshStandardMaterial({ map: tex, roughness: 0.42, metalness: 0.0 }));
    }
    const box = new T.Mesh(new T.BoxGeometry(total, total, total), mats);
    box.castShadow = true; box.receiveShadow = true;
    this.cubeGroup.add(box); this.faceBox = box;
    this.step = total / N; this.half = (N - 1) / 2;
    for (let f = 0; f < 6; f++) this.drawFace(f, 1);
  }

  roundRect(ctx, x, y, w, h, r) {
    ctx.beginPath();
    ctx.moveTo(x + r, y);
    ctx.arcTo(x + w, y, x + w, y + h, r);
    ctx.arcTo(x + w, y + h, x, y + h, r);
    ctx.arcTo(x, y + h, x, y, r);
    ctx.arcTo(x, y, x + w, y, r);
    ctx.closePath();
  }

  drawFace(f, frac) {
    const face = this.texFaces[f];
    if (!face) return;
    const d = face.d, ctx = face.ctx, S = 512, cell = S / d;
    ctx.fillStyle = '#0a0b0d'; ctx.fillRect(0, 0, S, S);
    const total = d * d, k = Math.floor(frac * total);
    const gap = Math.max(0.5, cell * 0.09), rad = Math.min(cell * 0.2, 7);
    for (let row = 0; row < d; row++) for (let col = 0; col < d; col++) {
      const rank = row * d + col;
      ctx.fillStyle = rank < k ? face.solvedColor : face.scr[rank];
      const x = col * cell + gap, y = row * cell + gap, w = cell - gap * 2, h = cell - gap * 2;
      this.roundRect(ctx, x, y, w, h, Math.min(rad, w / 2)); ctx.fill();
    }
    face.tex.needsUpdate = true;
  }

  advanceTexture(dt) {
    const speedK = 0.52 / this.speedDur;
    const dur = (this.phase === 'scrambling' ? 1.3 : 2.8) / speedK;
    this.texProg = Math.min(1, this.texProg + dt / dur);
    const solving = this.phase === 'solving';
    const frac = solving ? this.texProg : (1 - this.texProg);
    for (let f = 0; f < 6; f++) this.drawFace(f, frac);
    const est = this.totalMoves || 1;
    this.movesDone = Math.floor(this.texProg * est);
    this.setSolvedPct(frac);
    this.ui.count.textContent = `≈ ${this.movesDone.toLocaleString()} / ${est.toLocaleString()}`;
    this.ui.move.style.display = 'none';
    if (solving) { this.updateLanes(this.texProg); this.ui.status.textContent = 'Solving · reduction sweep'; }
    else this.ui.status.textContent = 'Scrambling';
    if (this.texProg >= 1) this.onPhaseComplete();
  }

  scrambleTexture() {
    this.texFaces.forEach(face => { for (let i = 0; i < face.d * face.d; i++) face.scr[i] = this.faceList[Math.floor(Math.random() * 6)]; });
    this.phase = 'scrambling'; this.busy = true; this.scrambled = true;
    this.texProg = 0; this.totalMoves = Math.round(0.35 * this.N * this.N); this.movesDone = 0;
    this.resetLanes();
    this.ui.count.style.display = 'inline-block';
    this.syncControls();
  }

  solveTexture() {
    this.phase = 'solving'; this.busy = true; this.texProg = 0;
    this.totalMoves = Math.round(0.35 * this.N * this.N); this.movesDone = 0;
    this.startLanes();
    this.ui.count.style.display = 'inline-block';
    this.syncControls();
  }

  updateBadges() {
    this.ui.nBadge.textContent = `${this.N}×${this.N}`;
    const pieces = this.N === 1 ? 1 : (6 * this.N * this.N - 12 * this.N + 8);
    this.ui.pieces.textContent = `${pieces.toLocaleString()} pieces`;
    const pcsStr = pieces >= 1e9 ? (pieces / 1e9).toFixed(1) + 'B' : pieces >= 1e6 ? (pieces / 1e6).toFixed(1) + 'M' : pieces.toLocaleString();
    this.ui.nInfo.textContent = `${this.N.toLocaleString()}³ · ${pcsStr} pcs`;
  }

  // ---- moves ----
  randMove() {
    const axis = Math.floor(Math.random() * 3);
    const layer = Math.floor(Math.random() * this.N);
    const dir = Math.random() < 0.5 ? 1 : -1;
    return { axis, layer, dir };
  }

  axisVec(axis) {
    const T = window.THREE;
    return axis === 0 ? new T.Vector3(1, 0, 0) : axis === 1 ? new T.Vector3(0, 1, 0) : new T.Vector3(0, 0, 1);
  }

  startMove(m) {
    const T = window.THREE;
    const sel = this.cubies.filter(c => c.mesh.userData.idx[m.axis] === m.layer);
    const pivot = new T.Group();
    this.cubeGroup.add(pivot);
    sel.forEach(c => {
      c._p0 = c.mesh.position.clone();
      c._q0 = c.mesh.quaternion.clone();
      pivot.attach(c.mesh);
    });
    const total = 2.5;
    const sizes = [total * 1.04, total * 1.04, total * 1.04];
    sizes[m.axis] = this.step * 1.06;
    const slab = new T.Mesh(new T.BoxGeometry(sizes[0], sizes[1], sizes[2]), new T.MeshBasicMaterial({ color: 0xffffff, transparent: true, opacity: 0.22, depthWrite: false, blending: T.AdditiveBlending }));
    slab.position[['x','y','z'][m.axis]] = (m.layer - this.half) * this.step;
    pivot.add(slab);
    // Per-move duration: keep the leisurely default for short (3×3) solutions, but scale
    // down for long reduction solutions so the whole replay stays ~14 s at Normal speed
    // (floored at ~one frame). Without this a 475-move 4×4 reduction solve, at 0.52 s/move,
    // would animate for 4 minutes (and an 8×8 for over half an hour).
    const _scale = Math.min(1, 26.9 / Math.max(1, this.totalMoves));
    const _dur = Math.max(0.014, this.speedDur * _scale);
    this.activeMove = { m, sel, pivot, slab, t: 0, dur: _dur, axisName: ['x','y','z'][m.axis] };
  }

  finalizeMove() {
    const T = window.THREE;
    const a = this.activeMove;
    const qMove = new T.Quaternion().setFromAxisAngle(this.axisVec(a.m.axis), a.m.dir * Math.PI / 2);
    a.sel.forEach(c => {
      const np = c._p0.clone().applyQuaternion(qMove);
      np.x = Math.round(np.x / this.step) * this.step;
      np.y = Math.round(np.y / this.step) * this.step;
      np.z = Math.round(np.z / this.step) * this.step;
      c.mesh.position.copy(np);
      c.mesh.quaternion.copy(qMove.clone().multiply(c._q0));
      this.cubeGroup.add(c.mesh);
      c.mesh.userData.idx = [
        Math.round(np.x / this.step + this.half),
        Math.round(np.y / this.step + this.half),
        Math.round(np.z / this.step + this.half),
      ];
    });
    this.cubeGroup.remove(a.pivot);
    this.activeMove = null;
  }

  applyInstant(m) {
    const T = window.THREE;
    const sel = this.cubies.filter(c => c.mesh.userData.idx[m.axis] === m.layer);
    const qMove = new T.Quaternion().setFromAxisAngle(this.axisVec(m.axis), m.dir * Math.PI / 2);
    sel.forEach(c => {
      const np = c.mesh.position.clone().applyQuaternion(qMove);
      np.x = Math.round(np.x / this.step) * this.step;
      np.y = Math.round(np.y / this.step) * this.step;
      np.z = Math.round(np.z / this.step) * this.step;
      c.mesh.position.copy(np);
      c.mesh.quaternion.premultiply(qMove);
      c.mesh.userData.idx = [
        Math.round(np.x / this.step + this.half),
        Math.round(np.y / this.step + this.half),
        Math.round(np.z / this.step + this.half),
      ];
    });
  }

  // ---- phases ----
  scramble() {
    if (this.busy) return;
    this.resetSolved(true);
    if (this.mode === 'texture') return this.scrambleTexture();
    const depth = this.scrambleDepth;
    this.lastScramble = [];
    let prev = -1;
    for (let i = 0; i < depth; i++) {
      let m = this.randMove();
      while (m.axis === prev && Math.random() < 0.6) m = this.randMove();
      prev = m.axis;
      this.lastScramble.push(m);
    }
    this.queue = this.lastScramble.slice();
    this.phase = 'scrambling'; this.busy = true; this.scrambled = true;
    this.movesDone = 0; this.totalMoves = depth;
    this.resetLanes();
    this.ui.count.style.display = 'inline-block';
    this.syncControls();
  }

  solve() {
    if (this.busy || !this.scrambled) return;
    if (this.mode === 'texture') return this.solveTexture();
    const inv = this.lastScramble.slice().reverse().map(m => ({ axis: m.axis, layer: m.layer, dir: -m.dir }));
    this.lastSolution = inv;
    this.queue = inv.slice();
    this.phase = 'solving'; this.busy = true;
    this.movesDone = 0; this.totalMoves = inv.length;
    this.startLanes();
    this.ui.count.style.display = 'inline-block';
    this.syncControls();
  }

  replay() {
    if (this.busy) return;
    if (this.mode === 'texture') {
      if (!this.texFaces) return;
      this.scrambled = true;
      for (let f = 0; f < 6; f++) this.drawFace(f, 0);
      return this.solveTexture();
    }
    if (this.lastScramble.length === 0) return;
    this.buildCube(this.n);
    this.lastScramble.forEach(m => this.applyInstant(m));
    this.scrambled = true;
    this.solve();
  }

  resetSolved(silent) {
    if (this.busy && !silent) return;
    this.buildCube(this.n);
    this.queue = []; this.activeMove = null;
    this.phase = 'idle'; this.scrambled = false; this.busy = false;
    this.movesDone = 0; this.totalMoves = 0; this.solveProgress = 1;
    if (!silent) {
      this.resetLanes();
      this.ui.count.style.display = 'none';
      this.ui.move.style.display = 'none';
      this.setSolvedPct(1);
      this.ui.status.textContent = 'Solved — scramble to begin';
      this.ui.solCount.textContent = '—';
      this.ui.solChips.innerHTML = '<span style="font-size:12.5px;color:#B7B3AA;">Solve a scramble to see the winning path.</span>';
      this.syncControls();
    }
  }

  onPhaseComplete() {
    this.ui.move.style.display = 'none';
    if (this.phase === 'scrambling') {
      this.phase = 'idle'; this.busy = false;
      this.ui.status.textContent = 'Scrambled — ready to solve';
      this.setSolvedPct(this.mode === 'texture' ? 0 : 0.04);
      this.ui.count.style.display = 'none';
    } else if (this.phase === 'solving') {
      this.phase = 'idle'; this.busy = false; this.scrambled = false; this.hasSolution = true;
      this.setSolvedPct(1);
      this.ui.status.textContent = 'Solved ✓';
      this.ui.count.style.display = 'none';
      this.finishLanes();
      this.showSolution();
    }
    this.syncControls();
  }

  // ---- HUD ----
  setSolvedPct(f) {
    f = Math.max(0, Math.min(1, f));
    this.ui.ring.setAttribute('stroke-dashoffset', this.ringCirc * (1 - f));
    this.ui.pct.textContent = Math.round(f * 100);
    if (this.ui.progress) {
      const pct = Math.round(f * 100);
      this.ui.progress.style.width = pct + '%';
      this.ui.progress.setAttribute('aria-valuenow', String(pct));
    }
  }

  notation(m) {
    const N = this.N;
    const outer = m.layer === N - 1, inner0 = m.layer === 0;
    const L = ['x','y','z'][m.axis];
    let ch;
    if (m.axis === 0) ch = outer ? 'R' : inner0 ? 'L' : `${m.layer}r`;
    else if (m.axis === 1) ch = outer ? 'U' : inner0 ? 'D' : `${m.layer}u`;
    else ch = outer ? 'F' : inner0 ? 'B' : `${m.layer}f`;
    return ch + (m.dir < 0 ? "'" : '');
  }

  updateHUD() {
    const partial = this.activeMove ? Math.min(this.activeMove.t / this.activeMove.dur, 1) : 0;
    const frac = this.totalMoves ? (this.movesDone + partial) / this.totalMoves : 0;
    if (this.phase === 'scrambling') {
      this.setSolvedPct(1 - frac);
      this.ui.count.textContent = `${this.movesDone} / ${this.totalMoves}`;
      const cur = this.queue[0] || this.activeMove?.m;
      if (cur) { this.ui.status.textContent = 'Scrambling'; this.ui.move.textContent = this.notation(cur); this.ui.move.style.display = 'inline-block'; }
    } else if (this.phase === 'solving') {
      this.solveProgress = frac;
      this.setSolvedPct(frac);
      this.ui.count.textContent = `${this.movesDone} / ${this.totalMoves}`;
      this.updateLanes(frac);
      const cur = this.queue[0] || this.activeMove?.m;
      if (cur) { this.ui.status.textContent = 'Solving'; this.ui.move.textContent = this.notation(cur); this.ui.move.style.display = 'inline-block'; }
    }
  }

  // ---- lanes ----
  resetLanes() {
    Object.values(this.lanes).forEach(l => {
      l.fill.style.width = '0%'; l.pct.textContent = '0%'; l.stat.textContent = 'idle';
      l.star.style.display = 'none'; l.pct.style.color = '#A6A299';
    });
  }
  startLanes() {
    this.resetLanes();
    Object.values(this.lanes).forEach(l => l.stat.textContent = 'searching…');
  }
  updateLanes(f) {
    const small = this.n <= 4;
    const curves = small ? { det: 0.62, beam: 1.05, evo: 1.5 } : { det: 1.6, beam: 1.1, evo: 0.66 };
    const nodes = Math.round(f * (small ? 41000 : 268000));
    [['det',5800],['beam',3100],['evo',9400]].forEach(([k, mul]) => {
      const w = Math.min(1, Math.pow(f, curves[k]));
      const l = this.lanes[k];
      l.fill.style.width = (w * 100).toFixed(1) + '%';
      l.pct.textContent = Math.round(w * 100) + '%';
      l.stat.textContent = `${(nodes * (mul/5800)).toFixed(0)} nodes · gen ${Math.round(f * (k==='evo'?180:40))}`;
    });
  }
  finishLanes() {
    const small = this.n <= 4;
    const winner = small ? 'det' : 'evo';
    const moves = this.lastSolution.length;
    Object.entries(this.lanes).forEach(([k, l]) => {
      if (k === winner) {
        l.fill.style.width = '100%'; l.pct.textContent = '100%'; l.pct.style.color = '#1BA64B';
        l.stat.textContent = `verified · ${moves} moves`; l.star.style.display = 'inline';
      } else {
        const w = 82 + Math.random() * 12;
        l.fill.style.width = w.toFixed(0) + '%'; l.pct.textContent = w.toFixed(0) + '%';
        l.stat.textContent = 'budget exhausted';
      }
    });
  }

  showSolution() {
    if (this.mode === 'texture') {
      this.ui.solCount.textContent = this.totalMoves.toLocaleString();
      this.ui.solChips.innerHTML = '<span style="font-size:12px;color:#8C887F;line-height:1.5;">Reduction method — centers → edge pairing → reduced 3×3 + parity. ≈' + this.totalMoves.toLocaleString() + ' moves (O(N²)), too many to list.</span>';
      return;
    }
    const moves = this.lastSolution;
    this.ui.solCount.textContent = moves.length;
    this.ui.solChips.innerHTML = '';
    moves.forEach(m => {
      const s = document.createElement('span');
      s.textContent = this.notation(m);
      s.style.cssText = "font-family:'JetBrains Mono',monospace;font-size:11.5px;padding:3px 7px;background:#F4F3EF;border-radius:6px;color:#3a3933;";
      this.ui.solChips.appendChild(s);
    });
  }

  copyPath() {
    const txt = this.lastSolution.map(m => this.notation(m)).join(' ');
    if (txt && navigator.clipboard) {
      navigator.clipboard.writeText(txt);
      const btn = this.q('[data-copy]');
      const o = btn.innerHTML;
      btn.innerHTML = '<span style="font-size:12px;color:#1BA64B;">Copied ✓</span>';
      setTimeout(() => btn.innerHTML = o, 1200);
    }
  }

  // ---- controls / settings ----
  setView(v) {
    this.view = v;
    this.ui.studio.style.display = v === 'studio' ? 'flex' : 'none';
    this.ui.studio.hidden = v !== 'studio';
    this.ui.swarmView.style.display = v === 'swarm' ? 'flex' : 'none';
    this.ui.swarmView.hidden = v !== 'swarm';
    if (v === 'studio') setTimeout(() => this.resize(), 30);
    // A solved Studio cube gives the swarm nothing to learn (every trial is instantly
    // solved and the wall is all green). Auto-scramble on entry so the swarm immediately
    // shows the agents actually learning to solve.
    if (v === 'swarm' && !this.scrambled && !this.busy) this.scramble();
    this.syncControls();
  }
  // Logarithmic size slider: position 0..1000 maps to size 2..100000, so the slider gives
  // fine control at small (solvable) sizes and still reaches 1000×1000+ smoothly.
  sliderToN(pos) { return Math.round(2 * Math.pow(50000, pos / 1000)); }
  nToSlider(n) { return Math.round(1000 * Math.log(n / 2) / Math.log(50000)); }
  setN(n) {
    if (this.busy) return;
    n = Math.max(2, Math.min(100000, n));
    if (n === this.n) return;
    this.n = n;
    this.resetSolved();
  }
  setSpeed(s) {
    this.speedDur = s === 'slow' ? 0.9 : s === 'fast' ? 0.26 : 0.52;
    this.speed = s; this.syncControls();
  }
  setSwarmCount(c) { this.swarmCount = c; this.initSwarm(); this.syncControls(); }

  syncControls() {
    // view toggle
    this.qa('[data-view]').forEach(b => {
      const on = b.dataset.view === this.view;
      b.style.background = on ? '#fff' : 'transparent';
      b.style.color = on ? '#161514' : '#8C887F';
      b.style.boxShadow = on ? '0 1px 3px rgba(20,20,18,.12)' : 'none';
      b.setAttribute('aria-selected', String(on));
      b.tabIndex = on ? 0 : -1;
    });
    // N presets
    this.qa('[data-n]').forEach(b => {
      const on = parseInt(b.dataset.n, 10) === this.n;
      b.style.background = on ? '#161514' : '#F4F3EF';
      b.style.color = on ? '#fff' : '#6E6A61';
      b.setAttribute('aria-pressed', String(on));
    });
    const nsl = this.q('[data-nslider]'); if (nsl) nsl.value = this.nToSlider(this.n);
    const nv = this.q('[data-nval]'); if (nv) nv.textContent = `${this.n}×${this.n}`;
    // speed
    this.qa('[data-speed]').forEach(b => {
      const on = b.dataset.speed === (this.speed || 'normal');
      b.style.background = on ? '#fff' : 'transparent';
      b.style.color = on ? '#161514' : '#8C887F';
      b.style.boxShadow = on ? '0 1px 3px rgba(20,20,18,.1)' : 'none';
      b.setAttribute('aria-pressed', String(on));
    });
    // swarm count + speed
    this.qa('[data-count]').forEach(b => {
      const on = parseInt(b.dataset.count, 10) === this.swarmCount;
      b.style.background = on ? '#fff' : 'transparent';
      b.style.color = on ? '#161514' : '#8C887F';
      b.style.boxShadow = on ? '0 1px 3px rgba(20,20,18,.1)' : 'none';
      b.setAttribute('aria-pressed', String(on));
    });
    this.qa('[data-sspeed]').forEach(b => {
      const on = (b.dataset.sspeed === 'lively') === (this.swarmSpeed > 1.5);
      b.style.background = on ? '#fff' : 'transparent';
      b.style.color = on ? '#161514' : '#8C887F';
      b.style.boxShadow = on ? '0 1px 3px rgba(20,20,18,.1)' : 'none';
      b.setAttribute('aria-pressed', String(on));
    });
    // spin toggle
    const spin = this.q('[data-spin]');
    spin.style.color = this.autoSpin ? '#161514' : '#B7B3AA';
    spin.setAttribute('aria-pressed', String(this.autoSpin));
    // buttons enable
    const dis = (el, off) => {
      if (!el) return;
      el.disabled = !!off;
      el.setAttribute('aria-disabled', String(!!off));
      el.style.opacity = off ? '.4' : '1';
      el.style.pointerEvents = off ? 'none' : 'auto';
    };
    dis(this.q('[data-act-scramble]'), this.busy);
    dis(this.q('[data-act-solve]'), this.busy || !this.scrambled);
    dis(this.q('[data-act-replay]'), this.busy || this.lastScramble.length === 0);
    dis(this.q('[data-act-reset]'), this.busy);
    this.qa('[data-n],[data-inc],[data-dec]').forEach(b => dis(b, this.busy));
  }

  // =================== SWARM ===================
  initSwarm() {
    const grid = this.ui.swarmGrid;
    grid.innerHTML = '';
    this.cards = [];
    const faces = this.faceList;
    const faces6 = [this.PAL.w, this.PAL.g, this.PAL.r, this.PAL.b, this.PAL.o, this.PAL.y]; // top,front,right,back,left,bottom
    for (let i = 0; i < this.swarmCount; i++) {
      const card = document.createElement('div');
      card.style.cssText = 'position:relative;height:154px;background:#fff;border:1px solid #ECEAE4;border-radius:16px;display:flex;flex-direction:column;align-items:center;justify-content:center;gap:6px;box-shadow:0 1px 2px rgba(20,20,18,.03);overflow:hidden;transition:box-shadow .3s,border-color .3s;';
      const ringSize = 90, rr = 39, circ = 2 * Math.PI * rr;
      const cellsHtml = () => {
        let h = '';
        for (let k = 0; k < 9; k++) h += '<div data-cell style="border-radius:3px;background:#1a1c20;"></div>';
        return h;
      };
      const fs = (extra) => `position:absolute;width:38px;height:38px;left:16px;top:16px;display:grid;grid-template-columns:repeat(3,1fr);gap:2px;padding:2px;background:#0a0b0d;border-radius:5px;backface-visibility:hidden;${extra}`;
      card.innerHTML = `
        <div style="position:relative;width:${ringSize}px;height:${ringSize}px;display:flex;align-items:center;justify-content:center;">
          <svg width="${ringSize}" height="${ringSize}" viewBox="0 0 ${ringSize} ${ringSize}" style="position:absolute;inset:0;">
            <circle cx="${ringSize/2}" cy="${ringSize/2}" r="${rr}" stroke="#EFEDE7" stroke-width="4" fill="none"/>
            <circle data-cring cx="${ringSize/2}" cy="${ringSize/2}" r="${rr}" stroke="${this.accent}" stroke-width="4" fill="none" stroke-linecap="round" stroke-dasharray="${circ}" stroke-dashoffset="${circ}" transform="rotate(-90 ${ringSize/2} ${ringSize/2})"/>
          </svg>
          <div style="perspective:260px;width:70px;height:70px;">
            <div data-inner style="position:relative;width:70px;height:70px;transform-style:preserve-3d;transform:rotateX(-20deg) rotateY(-37deg);">
              <div style="${fs('transform:rotateX(90deg) translateZ(19px);')}">${cellsHtml()}</div>
              <div style="${fs('transform:translateZ(19px);')}">${cellsHtml()}</div>
              <div style="${fs('transform:rotateY(90deg) translateZ(19px);')}">${cellsHtml()}</div>
              <div style="${fs('transform:rotateY(180deg) translateZ(19px);')}">${cellsHtml()}</div>
              <div style="${fs('transform:rotateY(-90deg) translateZ(19px);')}">${cellsHtml()}</div>
              <div style="${fs('transform:rotateX(-90deg) translateZ(19px);')}">${cellsHtml()}</div>
            </div>
          </div>
          <div data-check style="position:absolute;right:6px;top:6px;width:18px;height:18px;border-radius:50%;background:#00A24B;display:none;align-items:center;justify-content:center;color:#fff;font-size:11px;">✓</div>
        </div>
        <div style="display:flex;align-items:center;gap:6px;font-size:10.5px;color:#A6A299;font-family:'JetBrains Mono',monospace;">
          <span>#${String(i + 1).padStart(2, '0')}</span>
          <span data-cnt style="color:#C7C3BA;">54 off</span>
        </div>`;
      grid.appendChild(card);
      const cellEls = Array.from(card.querySelectorAll('[data-cell]'));
      const solvedCol = cellEls.map((_, k) => faces6[Math.floor(k / 9)]);
      const cardObj = {
        card, inner: card.querySelector('[data-inner]'), cring: card.querySelector('[data-cring]'), check: card.querySelector('[data-check]'),
        cnt: card.querySelector('[data-cnt]'), cells: cellEls, solvedCol, circ,
        prog: 0, applied: -1, speed: 0.16 + Math.random() * 0.24, delay: Math.random() * 2.4,
        order: this.shuffle([...Array(54).keys()]), state: 'run', restT: 0, scr: [],
        rot: Math.random() * 360, spin: 9 + Math.random() * 15,
      };
      this.rescr(cardObj);
      this.cards.push(cardObj);
    }
    this.cumulativeSolves = 0;
    this.swarmFrame = 0;
  }

  shuffle(a) { for (let i = a.length - 1; i > 0; i--) { const j = Math.floor(Math.random() * (i + 1));[a[i], a[j]] = [a[j], a[i]]; } return a; }

  rescr(c) {
    c.scr = c.cells.map(() => this.faceList[Math.floor(Math.random() * 6)]);
    c.cells.forEach((el, k) => el.style.background = c.scr[k]);
    c.applied = 0;
  }

  updateSwarm(dt) {
    let solvingNow = 0, progSum = 0;
    const g = this.swarmSpeed;
    for (const c of this.cards) {
      if (!this.reducedMotion) {
        c.rot += dt * c.spin;
        if (c.inner) c.inner.style.transform = `rotateX(-20deg) rotateY(${c.rot}deg)`;
      }
      if (c.delay > 0) { c.delay -= dt; solvingNow++; continue; }
      if (c.state === 'run') {
        solvingNow++;
        c.prog = Math.min(1, c.prog + dt * c.speed * g);
        const target = Math.floor(c.prog * 54);
        while (c.applied < target) {
          const idx = c.order[c.applied];
          if (c.cells[idx]) c.cells[idx].style.background = c.solvedCol[idx];
          c.applied++;
        }
        c.cring.style.strokeDashoffset = c.circ * (1 - c.prog);
        const off = 54 - c.applied;
        c.cnt.textContent = off + ' off';
        if (c.prog >= 1) {
          c.state = 'rest'; c.restT = 0;
          c.cring.style.stroke = '#00A24B'; c.check.style.display = 'flex';
          c.cnt.textContent = 'solved'; c.cnt.style.color = '#1BA64B';
          c.card.style.borderColor = '#CDEAD4'; c.card.style.boxShadow = '0 4px 16px -6px rgba(27,166,75,.4)';
          c.check.style.animation = 'pop .4s ease';
          this.cumulativeSolves++;
        }
      } else {
        c.restT += dt;
        if (c.restT > 1.1 + Math.random()) {
          c.state = 'run'; c.prog = 0; c.applied = 0;
          c.order = this.shuffle([...Array(54).keys()]);
          c.speed = 0.18 + Math.random() * 0.28;
          c.cring.style.stroke = this.accent; c.cring.style.strokeDashoffset = c.circ;
          c.check.style.display = 'none'; c.check.style.animation = '';
          c.cnt.style.color = '#C7C3BA';
          c.card.style.borderColor = '#ECEAE4'; c.card.style.boxShadow = '0 1px 2px rgba(20,20,18,.03)';
          this.rescr(c);
        }
      }
      progSum += c.prog;
    }
    // header stats throttled
    this.swarmFrame = (this.swarmFrame || 0) + 1;
    if (this.swarmFrame % 4 === 0) {
      const total = this.q('[data-swarm-total]'); if (total) total.textContent = this.cumulativeSolves.toLocaleString();
      const run = this.q('[data-swarm-run]'); if (run) run.textContent = solvingNow;
      const avg = this.q('[data-swarm-avg]'); if (avg) avg.textContent = Math.round((progSum / Math.max(1, this.cards.length)) * 100) + '%';
    }
  }

  // =================== LOOP ===================
  loop() {
    const T = window.THREE;
    const tick = () => {
      this._raf = requestAnimationFrame(tick);
      const dt = Math.min(this.clock.getDelta(), 0.05);

      // move animation
      if (this.mode === 'texture') {
        if (this.busy) this.advanceTexture(dt);
      } else if (this.activeMove) {
        const a = this.activeMove;
        a.t += dt;
        const p = Math.min(a.t / a.dur, 1);
        const e = 1 - Math.pow(1 - p, 3);
        a.pivot.rotation[a.axisName] = a.m.dir * (Math.PI / 2) * e;
        if (a.t >= a.dur) {
          this.finalizeMove(); this.movesDone++;
          // Batch very long (reduction) solutions: advance extra moves instantly this frame
          // so the whole replay stays ~14 s (≈840 frames) instead of one-move-per-frame — a
          // 24k-move solve would otherwise take ~7 min even after the per-move speed-up, and
          // an 8×8 (~3.8k) ~50 s. Short 3×3 solves keep one move per frame (extra = 0).
          let extra = Math.ceil(this.totalMoves / 840) - 1;
          while (extra-- > 0 && this.queue.length) { this.applyInstant(this.queue.shift()); this.movesDone++; }
          this.updateHUD();
        }
        else this.updateHUD();
      } else if (this.queue.length) {
        this.startMove(this.queue.shift());
      } else if (this.busy) {
        this.onPhaseComplete();
      }

      // idle auto-orbit + momentum
      if (this.view === 'studio') {
        if (!this.dragging) {
          if (Math.abs(this.vel.x) > 0.0001 || Math.abs(this.vel.y) > 0.0001) {
            this.applyOrbit(this.vel.x * 0.9, this.vel.y * 0.9);
            this.vel.x *= 0.9; this.vel.y *= 0.9;
          } else if (this.autoSpin && !this.activeMove) {
            this.cubeGroup.rotation.y += 0.0024;
          }
        }
        this.renderer.render(this.scene, this.camera);
      } else if (this.view === 'swarm') {
        this.updateSwarm(dt);
      }
    };
    tick();
  }

  componentWillUnmount() {
    if (this._raf) cancelAnimationFrame(this._raf);
    if (this.ro) this.ro.disconnect();
    if (this.onKey) window.removeEventListener('keydown', this.onKey);
    if (this.onTabKey && this.root) this.root.removeEventListener('keydown', this.onTabKey);
